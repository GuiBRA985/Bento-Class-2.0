        // ============================================================
        // STATE
        // ============================================================
        let state = {
            world: 'house',
            money: 1000000,
            satisfaction: 70,
            owned: {},
            vocab: {},
            currentRequest: null,
            streak: 0,
            correctCount: 0,
            totalWords: 0,
            startTime: Date.now(),
            log: [],
            activeCategory: 'build',
            quizActive: false,
            _quizWord: '',
            _quizCorrect: ''
        };

        const $ = id => document.getElementById(id);
        const $$ = sel => document.querySelectorAll(sel);

        // ============================================================
        // PERSISTENCE
        // ============================================================
        function saveProgress() {
            try {
                const data = {
                    world: state.world,
                    money: state.money,
                    satisfaction: state.satisfaction,
                    owned: state.owned,
                    vocab: state.vocab,
                    streak: state.streak,
                    correctCount: state.correctCount,
                    totalWords: state.totalWords,
                    startTime: state.startTime,
                    log: state.log.slice(-50)
                };
                localStorage.setItem('bentoLab_v2', JSON.stringify(data));
            } catch (_) {}
        }

        function loadProgress() {
            try {
                const raw = localStorage.getItem('bentoLab_v2');
                if (!raw) return;
                const data = JSON.parse(raw);
                Object.assign(state, data);
                if (!state.startTime) state.startTime = Date.now();
                if (!state.log) state.log = [];
                if (!state.owned) state.owned = {};
                if (!state.vocab) state.vocab = {};
            } catch (_) {}
        }

        function resetProgress() {
            if (!confirm('Resetar todo o progresso?')) return;
            localStorage.removeItem('bentoLab_v2');
            location.reload();
        }

        // ============================================================
        // RENDER
        // ============================================================
        function renderWorld() {
            $$('.world-card').forEach(el => {
                const w = el.dataset.world;
                el.classList.toggle('active', w === state.world);
            });

            const w = WORLDS[state.world];
            if (!w) return;

            $('panelTitle').textContent = w.label + ' Shop';
            $('panelText').textContent = w.desc;

            if (!state.owned) state.owned = {};
            if (!state.vocab) state.vocab = {};

            state.activeCategory = w.categories[0]?.id || 'build';
            renderTabs();
            renderShop();
            renderLot();
            renderVocab();
            renderLog();
            updateStats();
        }

        function renderTabs() {
            const w = WORLDS[state.world];
            const container = $('panelTabs');
            container.innerHTML = w.categories.map(c =>
                `<button class="${c.id === state.activeCategory ? 'active' : ''}" 
                                 onclick="setCategory('${c.id}')">${c.label}</button>`
            ).join('');
        }

        function setCategory(cat) {
            state.activeCategory = cat;
            renderTabs();
            renderShop();
        }

        function renderShop() {
            const w = WORLDS[state.world];
            const items = w.items.filter(i => i.cat === state.activeCategory);
            const container = $('panelShop');

            if (items.length === 0) {
                container.innerHTML = '<div style="color:var(--text-dim);padding:20px;text-align:center;">Nenhum item nesta categoria.</div>';
                return;
            }

            container.innerHTML = items.map(item => {
                const owned = !!state.owned[item.id];
                return `
                            <div class="shop-item ${owned ? 'owned' : ''}">
                                <div class="row">
                                    <span class="name">${item.emoji} ${item.name}</span>
                                    <span class="price">$${formatMoney(item.price)}</span>
                                </div>
                                <div class="desc">${item.desc}</div>
                                <div class="actions">
                                    <button class="btn-listen" onclick="speakText('${item.name.replace(/'/g, "\\'")}', 'en-US')">🔊 Ouvir</button>
                                    <button class="btn-buy" ${owned ? 'disabled' : ''} onclick="buyItem('${item.id}')">
                                        ${owned ? '✅ Adquirido' : '💰 Comprar'}
                                    </button>
                                </div>
                            </div>
                        `;
            }).join('');
        }

        function renderLot() {
            const w = WORLDS[state.world];
            const grid = $('lotGrid');
            const ownedIds = Object.keys(state.owned).filter(id =>
                w.items.some(i => i.id === id)
            );
            const items = ownedIds.map(id => w.items.find(i => i.id === id)).filter(Boolean);

            let html = '';
            if (items.length === 0) {
                html =
                    `<div class="lot-cell" style="grid-column:1/-1;min-height:120px;color:var(--text-dim);">🏗️ <br><span style="font-size:11px;">Compre itens para construir seu mundo</span></div>`;
            } else {
                const display = items.slice(0, 6);
                html = display.map(item =>
                    `<div class="lot-cell has-item">
                                <span class="emoji">${item.emoji}</span>
                                <span class="label">${item.name}</span>
                            </div>`
                ).join('');
                for (let i = display.length; i < 6; i++) {
                    html += `<div class="lot-cell" style="color:var(--text-dim);opacity:0.3;">✦</div>`;
                }
            }
            grid.innerHTML = html;
        }

        function renderVocab() {
            const tags = $('vocabTags');
            const words = Object.keys(state.vocab).sort();
            if (words.length === 0) {
                tags.innerHTML = '<span style="opacity:0.4;">Nenhuma palavra ainda</span>';
                return;
            }
            tags.innerHTML = words.map(w => `<span>${w}</span>`).join('');
            $('statWords').textContent = words.length;
        }

        function renderLog() {
            const container = $('logEntries');
            const entries = state.log.slice(-12);
            if (entries.length === 0) {
                container.innerHTML =
                    '<div style="opacity:0.4;font-style:italic;">Aguardando interações...</div>';
                return;
            }
            container.innerHTML = entries.map(e =>
                `<div><span class="entry-time">[${e.time}]</span> ${e.text}</div>`
            ).join('');
        }

        function renderStats() {
            $('statWords').textContent = Object.keys(state.vocab).length;
            $('statCorrect').textContent = state.correctCount;
            $('statStreak').textContent = state.streak;
            const mins = Math.floor((Date.now() - state.startTime) / 60000);
            $('statTime').textContent = mins;
        }

        function updateStats() {
            $('moneyBox').textContent = '$ ' + formatMoney(state.money);
            const sat = Math.max(0, Math.min(100, state.satisfaction));
            const fill = $('satFill');
            fill.style.width = sat + '%';
            fill.style.background = sat >= 70 ? '#34d399' : sat >= 40 ? '#fbbf24' : '#f87171';
            $('satLabel').textContent = (sat >= 70 ? '😊' : sat >= 40 ? '😐' : '😤') + ' ' + sat + '%';
            renderStats();
        }

        // ============================================================
        // GAME LOGIC
        // ============================================================
        function buyItem(itemId) {
            if (state.quizActive) return;

            const w = WORLDS[state.world];
            const item = w.items.find(i => i.id === itemId);
            if (!item || state.owned[itemId]) return;

            if (state.money < item.price) {
                setSpeech('💸', 'Parece que até ser rico tem limites. Não podemos pagar isso agora.');
                return;
            }

            state.money -= item.price;
            state.owned[itemId] = true;

            item.words.forEach(w => {
                if (!state.vocab[w]) {
                    state.vocab[w] = true;
                    state.totalWords++;
                }
            });

            addLog(`🛒 Comprou ${item.name} ($${formatMoney(item.price)})`);

            renderShop();
            renderLot();
            renderVocab();
            updateStats();

            if (state.currentRequest && state.currentRequest.need === itemId) {
                state.satisfaction = Math.min(100, state.satisfaction + 15);
                state.streak++;
                state.correctCount++;
                setSpeech('🎉', state.currentRequest.success);
                addLog(`✅ Cliente aprovou: ${state.currentRequest.success}`);
                state.currentRequest = null;
                triggerQuiz(item);
            } else {
                state.satisfaction = Math.max(0, state.satisfaction - 3);
                const complaint = w.complaints[Math.floor(Math.random() * w.complaints.length)];
                setSpeech('😤', complaint);
                addLog(`❌ Cliente reclamou: ${complaint}`);
                state.streak = 0;
            }

            updateStats();

            if (state.satisfaction <= 0) {
                clientLeaves();
                return;
            }

            if (!state.currentRequest) {
                setTimeout(() => newRequest(), 800);
            }
        }

        function newRequest() {
            if (state.quizActive) return;

            const w = WORLDS[state.world];
            const available = w.requests.filter(r => !state.owned[r.need]);
            if (available.length === 0) {
                setSpeech('🏆', 'Você completou todos os pedidos! Parabéns! Continue comprando itens para aprender mais.');
                return;
            }

            if (state.currentRequest) return;

            const req = available[Math.floor(Math.random() * available.length)];
            state.currentRequest = req;
            setSpeech('🧑‍💼', req.line);
            addLog(`📋 Novo pedido: ${req.line}`);
        }

        function clientLeaves() {
            setSpeech('🚪', 'O cliente está insatisfeito e vai embora. Tente novamente!');
            addLog('🚪 Cliente foi embora.');
            state.satisfaction = 0;
            updateStats();

            setTimeout(() => {
                state.satisfaction = 70;
                state.owned = {};
                state.currentRequest = null;
                state.streak = 0;
                renderWorld();
                updateStats();
                setSpeech('🔄', 'Novo cliente chegou! Atenda aos pedidos para aprender vocabulário.');
                setTimeout(() => newRequest(), 1000);
            }, 3000);
        }

        // ============================================================
        // QUIZ
        // ============================================================
        function triggerQuiz(item) {
            if (state.quizActive) return;
            state.quizActive = true;

            const overlay = $('quizOverlay');
            const title = $('quizTitle');
            const desc = $('quizDesc');
            const options = $('quizOptions');
            const feedback = $('quizFeedback');
            const closeBtn = $('quizClose');

            overlay.classList.add('show');
            feedback.className = 'quiz-feedback';
            feedback.style.display = 'none';
            closeBtn.style.display = 'none';

            const mainWord = item.words[0] || item.name;
            title.textContent = `🧠 O que significa "${mainWord}"?`;
            desc.textContent = 'Escolha a tradução correta:';

            const allWords = Object.keys(state.vocab).filter(w => w !== mainWord);
            const shuffled = allWords.sort(() => Math.random() - 0.5);
            let fakeOptions = shuffled.slice(0, 3);
            const w = WORLDS[state.world];
            while (fakeOptions.length < 3) {
                const extra = w.items.flatMap(i => i.words).filter(wd => !fakeOptions.includes(wd) && wd !== mainWord);
                if (extra.length === 0) break;
                fakeOptions.push(extra[Math.floor(Math.random() * extra.length)]);
            }

            const correctTranslation = item.name;
            const fakeTranslations = fakeOptions.map(wd => {
                const found = w.items.find(i => i.words.includes(wd));
                return found ? found.name : wd;
            });

            const allOpts = [
                { text: correctTranslation, correct: true },
                ...fakeTranslations.map(t => ({ text: t, correct: false }))
            ].sort(() => Math.random() - 0.5);

            options.innerHTML = allOpts.map(opt =>
                `<button onclick="answerQuiz(this, ${opt.correct}, '${mainWord}', '${correctTranslation}')">
                            ${opt.text}
                        </button>`
            ).join('');

            state._quizWord = mainWord;
            state._quizCorrect = correctTranslation;
        }

        function answerQuiz(btn, isCorrect, word, correctTranslation) {
            const options = $('quizOptions');
            const feedback = $('quizFeedback');
            const closeBtn = $('quizClose');

            options.querySelectorAll('button').forEach(b => b.disabled = true);

            if (isCorrect) {
                btn.classList.add('correct');
                feedback.className = 'quiz-feedback success show';
                feedback.textContent = `✅ Correto! "${word}" significa "${correctTranslation}". Muito bem!`;
                state.correctCount++;
                state.streak++;
            } else {
                btn.classList.add('wrong');
                options.querySelectorAll('button').forEach(b => {
                    if (b.textContent.trim() === correctTranslation) {
                        b.classList.add('correct');
                    }
                });
                feedback.className = 'quiz-feedback fail show';
                feedback.textContent = `❌ "${word}" significa "${correctTranslation}". Continue praticando!`;
                state.streak = 0;
            }

            renderStats();
            closeBtn.style.display = 'block';
            closeBtn.onclick = () => {
                $('quizOverlay').classList.remove('show');
                state.quizActive = false;
                if (!state.currentRequest && state.satisfaction > 0) {
                    setTimeout(() => newRequest(), 600);
                }
            };
        }

        // ============================================================
        // SPEECH & HELP
        // ============================================================
        function speakText(text, lang = 'en-US') {
            if (!text) return;
            window.speechSynthesis.cancel();
            const utter = new SpeechSynthesisUtterance(text);
            utter.lang = lang;
            utter.rate = 0.85;
            utter.pitch = 1;
            const voices = window.speechSynthesis.getVoices();
            const enVoice = voices.find(v => v.lang.startsWith('en'));
            if (enVoice) utter.voice = enVoice;
            window.speechSynthesis.speak(utter);
        }

        function setSpeech(emoji, text) {
            const avatar = $('clientAvatar');
            const speech = $('speechText');
            avatar.textContent = emoji || '🧑‍💼';
            speech.innerHTML = text;
            const englishMatch = text.match(/\b[a-z]{3,}\b/gi);
            if (englishMatch && englishMatch.length > 0) {
                speakText(text.replace(/[^a-zA-Z\s]/g, ''), 'en-US');
            }
        }

        function addLog(text) {
            const time = new Date().toLocaleTimeString('pt-BR', { hour: '2-digit', minute: '2-digit' });
            state.log.push({ time, text });
            if (state.log.length > 100) state.log.shift();
            renderLog();
        }

        function helpTip() {
            const tips = [
                '💡 Compre itens que o cliente pede para aumentar a satisfação.',
                '💡 Cada item ensina novas palavras em inglês.',
                '💡 Após comprar o item certo, você faz um quiz para fixar.',
                '💡 Mantenha a satisfação acima de 0 para o cliente não ir embora.',
                '💡 Explore todas as categorias para aprender mais vocabulário.'
            ];
            const tip = tips[Math.floor(Math.random() * tips.length)];
            setSpeech('💡', tip);
            addLog(`💡 Dica: ${tip}`);
        }

        function formatMoney(n) {
            return n.toLocaleString('pt-BR');
        }

        function selectWorld(worldId) {
            if (worldId === state.world) return;
            if (worldId !== 'house') {
                setSpeech('🔒', `O mundo "${WORLDS[worldId]?.label || worldId}" estará disponível em breve!`);
                return;
            }
            state.world = worldId;
            state.owned = {};
            state.currentRequest = null;
            state.satisfaction = 70;
            state.streak = 0;
            renderWorld();
            updateStats();
            setSpeech('🔄', `Bem-vindo ao ${WORLDS[worldId].label}! Atenda aos pedidos do cliente.`);
            setTimeout(() => newRequest(), 1000);
        }

        // ============================================================
        // EXPOSE GLOBALS
        // ============================================================
        window.selectWorld = selectWorld;
        window.setCategory = setCategory;
        window.buyItem = buyItem;
        window.newRequest = newRequest;
        window.speakText = speakText;
        window.helpTip = helpTip;
        window.resetProgress = resetProgress;
        window.answerQuiz = answerQuiz;

        // ============================================================
        // INIT
        // ============================================================
        function init() {
            loadProgress();
            renderWorld();
            renderStats();
            setTimeout(() => newRequest(), 1200);
            setInterval(saveProgress, 30000);
        }

        document.addEventListener('DOMContentLoaded', init);
    </script>
